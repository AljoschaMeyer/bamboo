\begin{align*}
  f(n) &= \begin{cases}
    n- 3^k &\mbox{if } n = \frac{3^k - 1}{2} \\
    n - \frac{3^{g(n)} - 1}{2} & \mbox{otherwise}\\
  \end{cases}\\
  g(n) &= \begin{cases}
    k &\mbox{if } n = \frac{3^k - 1}{2} \\
    g(n - \frac{3^{k - 1} - 1}{2}) & \mbox{if } \frac{3^{k - 1} - 1}{2} < n < \frac{3^k - 1}{2}
  \end{cases}\\
\end{align*}
